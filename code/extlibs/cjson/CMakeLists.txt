# minimal cmake file for cJSON

SET(SOURCES cJSON/cJSON.c cJSON/cJSON.h cJSON/cJSON_Utils.h cJSON/cJSON_Utils.c)

SOURCE_GROUP(cjson FILES ${SOURCES})

add_library(cjson STATIC ${SOURCES})

TARGET_INCLUDE_DIRECTORIES(cjson PUBLIC ${Nebula3_SOURCE_DIR}/code/extlibs/cjson/cJSON)
IF(MSVC)
	SET_PROPERTY(TARGET cjson APPEND PROPERTY COMPILE_FLAGS /wd"4996" )
ENDIF()
SET_TARGET_PROPERTIES(cjson PROPERTIES FOLDER "extlibs")
